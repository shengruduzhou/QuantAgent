/* 主力吸筹副图 V2：修复老公式变量问题，适合观察低位吸筹和高位风险 */

底线:0,COLOR808080;
中线:50,POINTDOT,COLORFFFFFF;
风险上沿:80,POINTDOT,COLORRED;

VAR1:=REF(LOW,1);
VAR2:=SMA(ABS(LOW-VAR1),3,1)/SMA(MAX(LOW-VAR1,0),3,1)*100;
VAR3:=EMA(IF(CLOSE>REF(CLOSE,1),VAR2*0.75,VAR2*1.25),3);
VAR4:=LLV(LOW,38);
VAR5:=HHV(VAR3,38);
VAR6:=IF(LLV(LOW,90)>0,1,0);

吸筹原:=EMA(IF(LOW<=VAR4,(VAR3+VAR5*2)/3,0),3)/6.18*VAR6;
主力吸筹:MIN(100,MAX(0,吸筹原)),COLORF0F0F0,LINETHICK2;

STICKLINE(主力吸筹>0,0,主力吸筹,2,0),COLOR0000FF;
STICKLINE(主力吸筹>10,0,主力吸筹,2,0),COLOR6600FF;
STICKLINE(主力吸筹>20,0,主力吸筹,2,0),COLORBB00FF;
STICKLINE(主力吸筹>35,0,主力吸筹,2,0),COLORFF00FF;
STICKLINE(主力吸筹>50,0,主力吸筹,2,0),COLORYELLOW;

RSV21:=(CLOSE-LLV(LOW,21))/(HHV(HIGH,21)-LLV(LOW,21)+0.0001)*100;
风险线:CEILING(SMA(RSV21,13,8)),COLORRED,LINETHICK2;

RSV27:=(CLOSE-LLV(LOW,27))/(HHV(HIGH,27)-LLV(LOW,27)+0.0001)*100;
K27:=SMA(RSV27,5,1);
D27:=SMA(K27,3,1);
涨跌线:MA(3*K27-2*D27,3),COLOR00FF00,LINETHICK2;

LC:=REF(CLOSE,1);
RSI6:=SMA(MAX(CLOSE-LC,0),6,1)/SMA(ABS(CLOSE-LC),6,1)*100;

V20:=MA(VOL,20);
VWAP5:=IF(SUM(VOL,5)>0,SUM(AMOUNT,5)/SUM(VOL,5)/100,CLOSE);
资金代理:=IF(CLOSE>=OPEN,
       (CLOSE-OPEN)/(HIGH-LOW+0.01)*AMOUNT/100000000,
      -(OPEN-CLOSE)/(HIGH-LOW+0.01)*AMOUNT/100000000);
资金流入率:=资金代理/(AMOUNT/100000000+0.01)*100;

低位吸筹:=主力吸筹>REF(主力吸筹,1) AND 主力吸筹>5 AND 风险线<45 AND CLOSE>VWAP5 AND 资金流入率>0;
吸筹确认:=CROSS(涨跌线,风险线) AND 涨跌线<55 AND 低位吸筹;
高位风险:=CROSS(风险线,涨跌线) AND 风险线>70;
卖压增强:=CROSS(85,RSI6) OR (风险线>80 AND 主力吸筹<REF(主力吸筹,1));

DRAWICON(吸筹确认,主力吸筹,1);
DRAWICON(高位风险,风险线,2);
DRAWICON(卖压增强,风险线,8);

DRAWTEXT(吸筹确认,主力吸筹,'吸');
DRAWTEXT(高位风险,风险线,'险');
DRAWTEXT(卖压增强,风险线,'卖压');

吸筹信号:IF(吸筹确认,1,0),NODRAW;
风险信号:IF(高位风险 OR 卖压增强,1,0),NODRAW;
