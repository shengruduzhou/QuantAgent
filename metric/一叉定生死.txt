/* 一叉定生死副图 V3：MACD+KDJ+VWAP+资金+筹码+大盘+风险扣分 */

N:=9;
M1:=3;
M2:=3;

DIFF:EMA(CLOSE,12)-EMA(CLOSE,26),COLORWHITE;
DEA:EMA(DIFF,9),COLORYELLOW;
MACD:=2*(DIFF-DEA);

零轴:0,COLOR808080;

STICKLINE(MACD>=0 AND MACD>=REF(MACD,1),0,MACD,4,0),COLORRED;
STICKLINE(MACD>=0 AND MACD<REF(MACD,1),0,MACD,4,0),COLORFF9900;
STICKLINE(MACD<0 AND MACD>=REF(MACD,1),0,MACD,4,0),COLOR00FFFF;
STICKLINE(MACD<0 AND MACD<REF(MACD,1),0,MACD,4,0),COLOR008000;

RSV:=(CLOSE-LLV(LOW,N))/(HHV(HIGH,N)-LLV(LOW,N)+0.0001)*100;
K:=SMA(RSV,M1,1);
D:=SMA(K,M2,1);
J:=3*K-2*D;

EMA13:=EMA(CLOSE,13);
EMA21:=EMA(CLOSE,21);
EMA55:=EMA(CLOSE,55);

V5:=MA(VOL,5);
V20:=MA(VOL,20);
V60:=MA(VOL,60);
A5:=MA(AMOUNT,5);
A20:=MA(AMOUNT,20);

VWAP5:=IF(SUM(VOL,5)>0,SUM(AMOUNT,5)/SUM(VOL,5)/100,CLOSE);
VWAP10:=IF(SUM(VOL,10)>0,SUM(AMOUNT,10)/SUM(VOL,10)/100,CLOSE);
VWAP20:=IF(SUM(VOL,20)>0,SUM(AMOUNT,20)/SUM(VOL,20)/100,CLOSE);

HSL:=IF(CAPITAL>0,VOL/CAPITAL*100,0);
HSL20:=MA(HSL,20);

TR1:=MAX(HIGH-LOW,MAX(ABS(HIGH-REF(CLOSE,1)),ABS(LOW-REF(CLOSE,1))));
ATR14:=MA(TR1,14);
波动率:=ATR14/CLOSE*100;

MB:=MA(CLOSE,20);
BOLL上:=MB+2*STD(CLOSE,20);
BOLL下:=MB-2*STD(CLOSE,20);
BOLL宽:=(BOLL上-BOLL下)/(MB+0.0001)*100;

前高20:=REF(HHV(HIGH,20),1);
前高60:=REF(HHV(HIGH,60),1);
前低20:=REF(LLV(LOW,20),1);

短支撑:=MAX(EMA13,VWAP5);
中支撑:=MAX(EMA21,VWAP10);
波段支撑:=MAX(EMA55,VWAP20);

多头:=CLOSE>EMA55 AND EMA13>EMA21 AND EMA21>EMA55;
空头:=CLOSE<EMA55 AND EMA13<EMA21 AND EMA21<EMA55;

MACD绿柱收敛:=MACD<0 AND MACD>REF(MACD,1) AND COUNT(MACD>REF(MACD,1),3)>=2;
MACD绿转红:=CROSS(MACD,0);
MACD红柱扩大:=MACD>0 AND MACD>REF(MACD,1) AND DIFF>DEA;
MACD主升:=MACD>0 AND MACD>REF(MACD,1)*1.08 AND DIFF>DEA AND DIFF>0;
MACD转弱:=MACD<REF(MACD,1) AND DIFF<REF(DIFF,1);

温和放量:=VOL>V5*1.10 AND VOL<V20*2.80;
主升量:=VOL>V20*1.80 AND AMOUNT>A20*1.50;
缩量回踩:=VOL<V20*1.20 AND LOW<=中支撑*1.02 AND CLOSE>=中支撑;

资金代理:=IF(CLOSE>=OPEN,
       (CLOSE-OPEN)/(HIGH-LOW+0.01)*AMOUNT/100000000,
      -(OPEN-CLOSE)/(HIGH-LOW+0.01)*AMOUNT/100000000);
资金流入率:=资金代理/(AMOUNT/100000000+0.01)*100;
资金连续流入:=COUNT(资金流入率>3,5)>=3 OR SUM(资金代理,5)>0;
资金明显流入:=资金流入率>8 AND AMOUNT>A5*1.10;

主力资金:=EMA(100*(CLOSE-LLV(LOW,34))/(HHV(HIGH,34)-LLV(LOW,34)+0.0001),3);
大盘资金:=EMA(100*(INDEXC-LLV(INDEXL,34))/(HHV(INDEXH,34)-LLV(INDEXL,34)+0.0001),3);
主力强:=主力资金>大盘资金 AND 主力资金>REF(主力资金,1);

筹码中位:=COST(50);
筹码承接:=CLOSE>筹码中位 AND WINNER(CLOSE)>REF(WINNER(CLOSE),1);
筹码过热:=WINNER(CLOSE)>0.88 AND CLOSE/EMA21>1.12;

相对强度:=IF(INDEXC>0,CLOSE/INDEXC,0);
相对走强:=相对强度>REF(相对强度,5) AND SLOPE(相对强度,5)>0;

大盘MA20:=MA(INDEXC,20);
大盘非弱:=INDEXC>大盘MA20 OR 大盘MA20>REF(大盘MA20,1);

价格未大涨:=CLOSE/REF(CLOSE,5)<1.08 AND CLOSE/REF(CLOSE,10)<1.15;
价格已拉开:=CLOSE/REF(CLOSE,5)>1.12 OR CLOSE/EMA21>1.15;
波动收敛:=波动率<MA(波动率,20)*0.95 OR BOLL宽<MA(BOLL宽,60)*0.85;
波动扩张:=波动率>REF(波动率,1) AND BOLL宽>REF(BOLL宽,1);

上破前高:=CLOSE>前高20+ATR14*0.10 AND CLOSE>OPEN AND VOL>V20*1.25;
强破前高:=CLOSE>前高60+ATR14*0.10 AND VOL>V20*1.50 AND AMOUNT>A20*1.30;

上影率:=(HIGH-MAX(CLOSE,OPEN))/(HIGH-LOW+0.01);
高位区:=CLOSE>前高20 OR CLOSE>BOLL上 OR 价格已拉开 OR 筹码过热;
放量滞涨:=高位区 AND VOL>V5*1.80 AND CLOSE<OPEN AND 上影率>0.30;
VWAP失守:=CLOSE<VWAP5 AND CLOSE<OPEN AND VOL>V5*1.10;
跌破波段:=CLOSE<波段支撑-ATR14*0.15 AND VOL>V20*1.15;
跌破前低:=CLOSE<前低20-ATR14*0.10 AND VOL>V20*1.10;

F趋势:=IF(多头,15,0);
F动量:=IF(MACD绿柱收敛 OR MACD绿转红 OR MACD红柱扩大,15,0);
F量能:=IF(温和放量 OR 主升量,12,0);
FVWAP:=IF(CLOSE>VWAP5 AND VWAP5>=VWAP10,12,0);
F资金:=IF(资金连续流入 OR 资金明显流入 OR 主力强,15,0);
F筹码:=IF(筹码承接,10,0);
F相对:=IF(相对走强,8,0);
F大盘:=IF(大盘非弱,8,0);
F波动:=IF(波动收敛 OR 波动扩张,5,0);

顶部分:=IF(放量滞涨,1,0)+IF(VWAP失守,1,0)+IF(筹码过热,1,0)+IF(MACD转弱 AND 高位区,1,0)+IF(跌破波段,2,0)+IF(跌破前低,2,0);
风险扣分:=IF(价格已拉开,8,0)+IF(筹码过热,10,0)+IF(HSL>HSL20*2.50 AND 价格已拉开,8,0)+IF(顶部分>=3,20,0)+IF(大盘非弱=0,12,0);

AI原始分:=F趋势+F动量+F量能+FVWAP+F资金+F筹码+F相对+F大盘+F波动-风险扣分;
AI分:=MAX(0,MIN(100,AI原始分));

一叉核心:=CROSS(DIFF,DEA) AND DIFF>0 AND CROSS(K,D) AND J>50;
预警原:=(CROSS(K,D) OR MACD绿柱收敛 OR MACD绿转红) AND CLOSE>EMA21 AND 多头 AND AI分>=55;
一叉原:=一叉核心 AND 多头 AND (温和放量 OR 主升量) AND CLOSE>VWAP5 AND AI分>=68 AND 顶部分<=1 AND 大盘非弱;
强攻原:=一叉原 AND (上破前高 OR 强破前高 OR 主升量) AND 价格已拉开=0 AND AI分>=75;
低吸原:=多头 AND 价格未大涨 AND 缩量回踩 AND 资金连续流入 AND MACD绿柱收敛 AND AI分>=60 AND 顶部分<=1;

死叉原:=CROSS(DEA,DIFF) AND J<60;
转弱原:=CLOSE<EMA21 AND MACD转弱 AND 顶部分>=2;
清仓原:=跌破波段 OR 跌破前低 OR (VWAP失守 AND 顶部分>=3) OR 大盘非弱=0;

一叉:=一叉原 AND REF(一叉原,1)=0;
预警:=预警原 AND 一叉=0 AND REF(预警原,1)=0;
强攻:=强攻原 AND REF(强攻原,1)=0;
低吸:=低吸原 AND REF(低吸原,1)=0;

死叉:=死叉原 AND REF(死叉原,1)=0;
转弱:=转弱原 AND REF(转弱原,1)=0;
清仓:=清仓原 AND REF(清仓原,1)=0;

底背:=LLV(CLOSE,20)<REF(LLV(CLOSE,20),5) AND J>REF(J,5) AND J<35 AND MACD>REF(MACD,1) AND 资金流入率>0;
顶背:=HHV(CLOSE,20)>REF(HHV(CLOSE,20),5) AND J<REF(J,5) AND J>65 AND MACD<REF(MACD,1);

买入候选:=一叉 OR 强攻 OR 低吸;
卖出候选:=死叉 OR 转弱 OR 清仓 OR 顶背;

回测买点:=买入候选 AND 卖出候选=0 AND COUNT(买入候选,5)=1;
回测卖点:=卖出候选 AND COUNT(卖出候选,3)=1;

DRAWICON(预警,MACD,7);
DRAWICON(一叉,MACD,1);
DRAWICON(强攻,MACD,9);
DRAWICON(低吸,MACD,23);
DRAWICON(死叉,MACD,2);
DRAWICON(转弱,MACD,8);
DRAWICON(清仓,MACD,15);
DRAWICON(底背,MACD,1);
DRAWICON(顶背,MACD,2);

DRAWTEXT(预警,MACD,'预');
DRAWTEXT(一叉,MACD,'一叉');
DRAWTEXT(强攻,MACD,'强');
DRAWTEXT(低吸,MACD,'低吸');
DRAWTEXT(死叉,MACD,'死');
DRAWTEXT(转弱,MACD,'弱');
DRAWTEXT(清仓,MACD,'清');

AI量化分:AI分,NODRAW;
顶部评分:顶部分,NODRAW;
资金流入率值:资金流入率,NODRAW;
买入信号:IF(回测买点,1,0),NODRAW;
卖出信号:IF(回测卖点,1,0),NODRAW;
