/* 成交量复合炮 V2：量均线 + VWAP + 量价趋势 + 回测信号 */

VOLUME:VOL,VOLSTICK;

MAVOL1:MA(VOL,1)*1.2,COLORWHITE;
MAVOL2:MA(VOL,2)*1.2,COLORYELLOW;
MAVOL5:MA(VOL,5)*1.2,COLOR00FFFF;
MAVOL13:MA(VOL,13)*1.2,COLORFF00FF;
MAVOL21:MA(VOL,21)*1.2,COLOR0080FF;
MAVOL34:MA(VOL,34)*1.2,COLOR808080;
MAVOL55:MA(VOL,55)*1.2,COLOR00FF00;
MAVOL120:MA(VOL,120)*2,COLORRED;
MAVOL180:MA(VOL,180)*2,COLORRED,LINETHICK3;

V5:=MA(VOL,5);
V20:=MA(VOL,20);
V60:=MA(VOL,60);
A20:=MA(AMOUNT,20);

VWAP5:=IF(SUM(VOL,5)>0,SUM(AMOUNT,5)/SUM(VOL,5)/100,CLOSE);
VWAP10:=IF(SUM(VOL,10)>0,SUM(AMOUNT,10)/SUM(VOL,10)/100,CLOSE);

RSV27:=(CLOSE-LLV(LOW,27))/(HHV(HIGH,27)-LLV(LOW,27)+0.0001)*100;
K27:=SMA(RSV27,5,1);
D27:=SMA(K27,3,1);
量价趋势:=MA(3*K27-2*D27,5);

HSL:=IF(CAPITAL>0,VOL/CAPITAL*100,0);
HSL20:=MA(HSL,20);

温和放量:=VOL>V5*1.10 AND VOL<V20*2.80;
主升量:=VOL>V20*1.80 AND AMOUNT>A20*1.50;
异常爆量:=VOL>V20*3.50 AND CLOSE<OPEN;
量均抬升:=V5>REF(V5,1) AND REF(V5,1)>REF(V5,2);
价在VWAP上:=CLOSE>VWAP5 AND VWAP5>=VWAP10;

关注:=CROSS(MAVOL1,MAVOL2) AND MAVOL2<MAVOL180*0.90 AND UPNDAY(EMA(CLOSE,13),2) AND CLOSE>VWAP5;
金箭:=CROSS(MAVOL1,MAVOL180) AND UPNDAY(MAVOL180,2) AND 价在VWAP上;
金炮:=UPNDAY(MAVOL1,3) AND UPNDAY(MAVOL21,2) AND UPNDAY(MAVOL34,2)
      AND UPNDAY(MAVOL55,1) AND UPNDAY(MAVOL120,1)
      AND UPNDAY(量价趋势,3) AND 主升量 AND 价在VWAP上;
出货:=异常爆量 OR (VOL>V20*2.50 AND CLOSE<OPEN AND HIGH/CLOSE>1.035 AND HSL>HSL20*1.80);

DRAWTEXT(关注,MAVOL1*1.20,'关注'),COLORYELLOW;
DRAWTEXT(金箭,MAVOL1*1.30,'金箭'),COLORYELLOW;
DRAWTEXT(金炮,MAVOL1*1.40,'金炮'),COLORFF00FF;
DRAWTEXT(出货,MAVOL1*1.50,'出货'),COLOR00FF00;

DRAWICON(关注,MAVOL1*1.20,7);
DRAWICON(金箭,MAVOL1*1.30,1);
DRAWICON(金炮,MAVOL1*1.40,9);
DRAWICON(出货,MAVOL1*1.50,2);

量能分:=IF(量均抬升,20,0)+IF(温和放量,20,0)+IF(主升量,25,0)+IF(价在VWAP上,20,0)+IF(量价趋势>REF(量价趋势,1),15,0)-IF(出货,40,0);
量能评分:MAX(0,MIN(100,量能分)),NODRAW;

回测量买:IF((关注 OR 金箭 OR 金炮) AND 出货=0,1,0),NODRAW;
回测量卖:IF(出货,1,0),NODRAW;
